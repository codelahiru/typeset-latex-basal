  {
    \hbox_set:Nn \l_tmpa_box { \num {#1} }
    \box_show:N \l_tmpa_box
    \hbox_set:Nn \l_tmpa_box { $ \num {#1} $ }
    \box_show:N \l_tmpa_box
    \group_begin:
      \keys_set:nn { siunitx } { mode = text }
      \hbox_set:Nn \l_tmpa_box { \num {#1} }
      \box_show:N \l_tmpa_box
      \hbox_set:Nn \l_tmpa_box { $ \num {#1} $ }
      \box_show:N \l_tmpa_box
    \group_end:
  }
